# Adding to array
    arr1 = array([1,2,3,4,5])

    arr2 = array([6,2,3,4,5])

    arr3 = arr1 + arr2

    print(arr3)

# Sin of the array
    arr1 = array([1,2,3,4,5])
    print(sin(arr1))

# Cos of the array
    arr1 = array([1,2,3,4,5])
    print(cos(arr1))

# Log of the array
    arr1 = array([1,2,3,4,5])
    print(log(arr1))

# Sqrt of the array
    arr1 = array([1,2,3,4,5])
    print(sqrt(arr1))

# Sum of the array
    arr1 = array([1,2,3,4,5])
    print(sum(arr1))

# To find min value
    arr1 = array([1,2,3,4,5])
    print(min(arr1))

# To find max value
    arr1 = array([1,2,3,4,5])
    print(max(arr1))
